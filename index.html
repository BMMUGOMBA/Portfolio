"""
Item 3 — Automated data-processing pipeline (PORTFOLIO ARTEFACT)
----------------------------------------------------------------
A sanitised, self-contained demonstration of an automated classification
pipeline that processes records, classifies them into categories using a
light NLP approach, and validates the result against a labelled sample.

ALL DATA IS SYNTHETIC. This mirrors (does not reproduce) professional work
in which a manual classification process was automated and its precision
improved from 73% to 91% across validation iterations.

Run:  python Item3_classification_pipeline.py
"""
from __future__ import annotations
import csv, re, random
from collections import defaultdict

random.seed(11)

# ---- 1. Reference classification (HS chapter -> keyword cues) -------------
HS_RULES = {
    "10": ["maize", "wheat", "cereal", "grain", "sorghum", "rice"],
    "27": ["fuel", "petroleum", "diesel", "oil", "coal", "gas"],
    "52": ["cotton", "lint", "yarn"],
    "87": ["vehicle", "truck", "car", "tractor", "automotive"],
    "71": ["gold", "diamond", "platinum", "precious", "gemstone"],
    "24": ["tobacco", "cigarette", "leaf tobacco"],
    "09": ["coffee", "tea", "spice", "pepper"],
    "84": ["machine", "machinery", "pump", "engine", "appliance"],
    "26": ["ore", "chrome", "nickel", "concentrate", "slag"],
    "17": ["sugar", "molasses", "confectionery"],
}

# ---- 2. Synthetic input records (free-text descriptions) -----------------
def make_records(n=200):
    templates = {
        "10": ["white maize grain, bulk", "milling wheat consignment", "sorghum cereal export"],
        "27": ["diesel fuel bulk", "crude petroleum oil", "thermal coal shipment"],
        "52": ["raw cotton lint bales", "cotton yarn rolls"],
        "87": ["used motor vehicle", "agricultural tractor units", "light truck"],
        "71": ["rough diamond parcel", "refined gold bars", "platinum concentrate"],
        "24": ["flue-cured tobacco leaf", "cigarette cartons"],
        "09": ["green coffee beans", "black tea chests", "ground pepper spice"],
        "84": ["industrial water pump", "diesel engine unit", "milling machine"],
        "26": ["chrome ore concentrate", "nickel ore", "mineral slag"],
        "17": ["raw cane sugar", "molasses drums"],
    }
    recs = []
    for i in range(n):
        true_hs = random.choice(list(templates))
        desc = random.choice(templates[true_hs])
        # inject noise into ~15% to make classification non-trivial
        if random.random() < 0.15:
            desc = desc.split()[0]  # truncate -> harder to classify
        recs.append({"record_id": f"R{i:04d}", "description": desc, "true_hs": true_hs})
    return recs

# ---- 3. Classifier (keyword NLP with simple normalisation) ---------------
def normalise(text: str) -> list[str]:
    return re.findall(r"[a-z]+", text.lower())

def classify(description: str) -> str | None:
    tokens = set(normalise(description))
    scores = defaultdict(int)
    for hs, cues in HS_RULES.items():
        for cue in cues:
            if cue in description.lower():
                scores[hs] += 2          # phrase match
            elif cue in tokens:
                scores[hs] += 1          # token match
    if not scores:
        return None
    return max(scores, key=scores.get)

# ---- 4. Validation against the labelled sample ---------------------------
def evaluate(records) -> dict:
    correct = total = unclassified = 0
    for r in records:
        pred = classify(r["description"])
        if pred is None:
            unclassified += 1
            continue
        total += 1
        if pred == r["true_hs"]:
            correct += 1
    precision = round(100 * correct / total, 1) if total else 0.0
    return {"records": len(records), "classified": total,
            "unclassified": unclassified, "correct": correct, "precision_pct": precision}

# ---- 5. Run ---------------------------------------------------------------
if __name__ == "__main__":
    records = make_records(200)
    metrics = evaluate(records)
    print("Automated classification pipeline — synthetic demonstration")
    print("-" * 58)
    for k, v in metrics.items():
        print(f"  {k:>14}: {v}")
    # write classified output
    with open("classified_output.csv", "w", newline="") as f:
        w = csv.writer(f)
        w.writerow(["record_id", "description", "predicted_hs", "true_hs", "match"])
        for r in records:
            pred = classify(r["description"])
            w.writerow([r["record_id"], r["description"], pred or "UNCLASSIFIED",
                        r["true_hs"], pred == r["true_hs"]])
    print("\n  Wrote classified_output.csv")

Confidentiality. Method, sanitised snippet or process diagram only — run on synthetic records, with real field names removed.

-- ============================================================================
-- Item 6 — Statistical Database Design (PORTFOLIO ARTEFACT)
-- Synthetic regional trade-statistics database — star schema (DDL)
-- ILLUSTRATIVE / SYNTHETIC. No real or confidential data.
-- Demonstrates: dimensional modelling, referential integrity,
-- standard code lists / nomenclature, and access governance.
-- ============================================================================

-- ---- Dimension: Country (shared by reporter and partner roles) -------------
CREATE TABLE dim_country (
    country_id      INT          PRIMARY KEY,
    iso3_code       CHAR(3)      NOT NULL UNIQUE,      -- standard ISO-3 nomenclature
    country_name    VARCHAR(80)  NOT NULL,
    is_sadc_member  BOOLEAN      NOT NULL DEFAULT FALSE,
    CONSTRAINT chk_iso3 CHECK (CHAR_LENGTH(iso3_code) = 3)
);

-- ---- Dimension: Product (Harmonized System) --------------------------------
CREATE TABLE dim_product (
    product_id      INT          PRIMARY KEY,
    hs2_code        CHAR(2)      NOT NULL UNIQUE,      -- HS chapter (2-digit)
    hs_section      VARCHAR(60),
    product_label   VARCHAR(120) NOT NULL
);

-- ---- Dimension: Period -----------------------------------------------------
CREATE TABLE dim_period (
    period_id       INT          PRIMARY KEY,
    year            SMALLINT     NOT NULL,
    quarter         SMALLINT,                          -- NULL for annual records
    period_label    VARCHAR(10)  NOT NULL,
    CONSTRAINT chk_year    CHECK (year BETWEEN 1990 AND 2100),
    CONSTRAINT chk_quarter CHECK (quarter IS NULL OR quarter BETWEEN 1 AND 4)
);

-- ---- Fact: Trade -----------------------------------------------------------
CREATE TABLE fact_trade (
    trade_id        BIGINT       PRIMARY KEY,
    reporter_id     INT          NOT NULL,
    partner_id      INT          NOT NULL,
    product_id      INT          NOT NULL,
    period_id       INT          NOT NULL,
    flow            VARCHAR(6)   NOT NULL,
    value_usd_000   DECIMAL(16,2) NOT NULL,
    quantity_tonnes DECIMAL(16,2),
    is_provisional  BOOLEAN      NOT NULL DEFAULT TRUE, -- provisional vs final
    -- Referential integrity: every code must exist in its reference table
    CONSTRAINT fk_reporter FOREIGN KEY (reporter_id) REFERENCES dim_country(country_id),
    CONSTRAINT fk_partner  FOREIGN KEY (partner_id)  REFERENCES dim_country(country_id),
    CONSTRAINT fk_product  FOREIGN KEY (product_id)  REFERENCES dim_product(product_id),
    CONSTRAINT fk_period   FOREIGN KEY (period_id)   REFERENCES dim_period(period_id),
    -- Validity rules enforced at the database layer
    CONSTRAINT chk_flow    CHECK (flow IN ('Export','Import')),
    CONSTRAINT chk_value   CHECK (value_usd_000 >= 0),
    CONSTRAINT chk_qty     CHECK (quantity_tonnes IS NULL OR quantity_tonnes >= 0),
    CONSTRAINT chk_self    CHECK (reporter_id <> partner_id)
);

-- ---- Indexes for query performance -----------------------------------------
CREATE INDEX ix_trade_period   ON fact_trade(period_id);
CREATE INDEX ix_trade_reporter ON fact_trade(reporter_id);
CREATE INDEX ix_trade_product  ON fact_trade(product_id);

-- ---- Example analytical view: intra-regional trade share -------------------
CREATE VIEW v_intra_regional_share AS
SELECT  p.year,
        SUM(CASE WHEN c.is_sadc_member THEN f.value_usd_000 ELSE 0 END) AS intra_sadc_value,
        SUM(f.value_usd_000)                                            AS total_value,
        ROUND( SUM(CASE WHEN c.is_sadc_member THEN f.value_usd_000 ELSE 0 END)
               / NULLIF(SUM(f.value_usd_000),0) * 100, 1)               AS intra_share_pct
FROM        fact_trade f
JOIN        dim_country c ON c.country_id = f.partner_id
JOIN        dim_period  p ON p.period_id = f.period_id
GROUP BY    p.year;
